// ============================================= // CAIXA CÍRCULO DE ORAÇÃO — Google Apps Script // ============================================= function doGet(e) { const p = e.parameter; const action = p.action; let result; try { if (action === 'getDados') result = getDados(); else if (action === 'addCulto') result = addCulto(p); else if (action === 'addSaida') result = addSaida(p); else if (action === 'addFiado') result = addFiado(p); else if (action === 'setCaixa') result = setCaixa(p); else if (action === 'deleteRow') result = deleteRow(p.sheet, parseInt(p.index)); else result = { error: 'Ação desconhecida: ' + action }; } catch(err) { result = { error: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // Converte Date para string DD/MM/AAAA function fmtData(val) { if (!val) return ''; if (Object.prototype.toString.call(val) === '[object Date]') { var d = Utilities.formatDate(val, Session.getScriptTimeZone(), 'dd/MM/yyyy'); return d; } return String(val); } function getDados() { const ss = SpreadsheetApp.getActiveSpreadsheet(); return { cultos: getSheet(ss, 'Cultos', [0]), saidas: getSheet(ss, 'Saidas', [0]), caixa: getCaixa(ss), fiados: getSheet(ss, 'Fiados', []), }; } // colsDatas: índices das colunas que são data function getSheet(ss, nome, colsDatas) { const sh = ss.getSheetByName(nome); if (!sh) return []; const rows = sh.getDataRange().getValues(); rows.shift(); return rows.map(function(r, i) { var vals = r.map(function(v, col) { if (colsDatas.indexOf(col) >= 0) return fmtData(v); return v; }); return { _index: i + 2, values: vals }; }); } function getCaixa(ss) { const sh = ss.getSheetByName('Caixa'); if (!sh) return { conta: 0, dinheiro: 0 }; return { conta: sh.getRange('B2').getValue(), dinheiro: sh.getRange('B3').getValue(), }; } function addCulto(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Cultos'); sh.appendRow([p.data, parseFloat(p.valor)]); return { ok: true }; } function addSaida(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Saidas'); sh.appendRow([p.data, p.descricao, parseFloat(p.valor)]); return { ok: true }; } function addFiado(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Fiados'); sh.appendRow([p.nome, parseFloat(p.valor)]); return { ok: true }; } function setCaixa(p) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Caixa'); sh.getRange('B2').setValue(parseFloat(p.conta)); sh.getRange('B3').setValue(parseFloat(p.dinheiro)); return { ok: true }; } function deleteRow(sheetName, rowIndex) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); sh.deleteRow(rowIndex); return { ok: true }; } function setup() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sh = ss.getSheetByName('Cultos') || ss.insertSheet('Cultos'); sh.clearContents(); sh.appendRow(['Data', 'Valor']); sh.appendRow(['07/05/2026', 65.70]); sh.appendRow(['14/05/2026', 156.50]); sh.appendRow(['21/05/2026', 112.00]); sh.appendRow(['28/05/2026', 129.80]); sh = ss.getSheetByName('Saidas') || ss.insertSheet('Saidas'); sh.clearContents(); sh.appendRow(['Data', 'Descrição', 'Valor']); sh.appendRow(['08/05/2026', 'Pregadora Elaine', 150.00]); sh.appendRow(['21/05/2026', 'Pregadora Maria de Lurdes', 150.00]); sh.appendRow(['28/05/2026', 'Pregadora Regina', 150.00]); sh.appendRow(['08/05/2026', 'Papel pão', 21.90]); sh.appendRow(['27/05/2026', 'Envelopes', 12.90]); sh = ss.getSheetByName('Caixa') || ss.insertSheet('Caixa'); sh.clearContents(); sh.appendRow(['Tipo', 'Valor']); sh.appendRow(['Em conta', 3935.36]); sh.appendRow(['Em dinheiro', 2099.00]); sh = ss.getSheetByName('Fiados') || ss.insertSheet('Fiados'); sh.clearContents(); sh.appendRow(['Nome', 'Valor']); sh.appendRow(['Juninho', 15.00]); sh.appendRow(['Claúdia (Vanessa)', 15.00]); sh.appendRow(['Hélida', 15.00]); sh.appendRow(['Mirian Mazali', 15.00]); sh.appendRow(['Leila', 30.00]); sh.appendRow(['Angélica', 30.00]); Logger.log('Setup concluído!'); }
